  authentication:
    description: "Allow users to create accounts and log in."
    subtasks:
      - Create User model
      - Registration API
      - Login API

      - JWT authentication
      - Protected routes middleware

    description: "Display available cloud kitchens to customers."
    subtasks:
      - Create Kitchen model
      - Kitchen listing API
      - Kitchen detail API
      - Display kitchen name
      - Display food categories
      - Display ratings
      - Display delivery area

    description: "Enable customers to pay for orders."
    subtasks:
      - Setup Razorpay account
      - Create Razorpay order API
      - Frontend payment integration
      - Payment success verification
      - Store transaction details
      - Update order status after successful payment

  worker_registration:

    subtasks:

      - Worker registration form
      - Upload identity proof
      - Upload driving license
      - Store worker details
      - Mark status as Pending Review

    description: "Allow cloud kitchens to apply for listing."
    subtasks:
      - Kitchen registration form
      - Upload kitchen documents
      - Upload FSSAI license
      - Upload GST details
      - Store application data
      - Set status to Under Review

  admin_review:
    description: "Admin reviews kitchens and workers."
    subtasks:
      - View pending kitchens
      - View pending workers
      - Approve application
      - Reject application
      - Update application status
